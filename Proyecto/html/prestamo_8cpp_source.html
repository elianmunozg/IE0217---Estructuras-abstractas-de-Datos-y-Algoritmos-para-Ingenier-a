24Prestamo::Prestamo(double monto, int cuotas, double tasa, std::string tipo, std::string id)
25 : montoPrestamo(monto), cantidadCuotas(cuotas), tasaInteres(tasa), tipoPrestamo(tipo), idPrestamo(id){
26 // Calcula el pago mensual usando la fórmula de amortización francesa
27 double tasaMensual = tasaInteres / 12 / 100;
28 double cuotaMensual = (montoPrestamo * tasaMensual) / (1 - pow(1 + tasaMensual, -cantidadCuotas));
29
30 double saldoRestante = montoPrestamo;
31 for (int i = 1; i <= cantidadCuotas; ++i) {
32 double intereses = saldoRestante * tasaMensual;
33 double capital = cuotaMensual - intereses;
34 saldoRestante -= capital;
35

189void Prestamo::generarReporte(const std::string& nombreArchivo) const {
190 std::ofstream archivoReporte(nombreArchivo);
191 if (!archivoReporte.is_open()) {
192 std::cerr << "No se pudo abrir el archivo para escribir el reporte del préstamo." << std::endl;
193 return;
194 }
195
196 archivoReporte << "Reporte de préstamo para ID: " << idPrestamo << "\n\n";
197 archivoReporte << "Detalle de Pagos Realizados:\n";
198 archivoReporte << "Cuota\tCapital\tIntereses\tTotal Pagado\n";
199 archivoReporte << "--------------------------------------------------\n";
200
201 double totalCapital = 0;
202 double totalIntereses = 0;
203 int cuotaNum = 1; // Iniciar el contador de cuotas
204 for (const auto& pago : pagos) {
205 double totalPorCuota = pago.first + pago.second; // Suma de capital e intereses para esta cuota
206 archivoReporte << cuotaNum << "\t" // Imprimir el número de cuota
207 << std::fixed << std::setprecision(2) << pago.first << "\t"
208 << std::fixed << std::setprecision(2) << pago.second << "\t"
209 << std::fixed << std::setprecision(2) << totalPorCuota << "\n"; // Total pagado en la cuota
210 totalCapital += pago.first; // Suma al total del capital
211 totalIntereses += pago.second; // Suma al total de intereses
212 cuotaNum++; // Incrementar el número de cuota para la siguiente iteración
213 }
214
215 // Calcular el saldo restante del préstamo
216 double saldoRestante = montoPrestamo - totalCapital;
217
218 // Resumen al final del reporte
219 archivoReporte << "\nResumen:\n";
220 archivoReporte << "Total pagado hacia el capital: $" << std::fixed << std::setprecision(2) << totalCapital << "\n";
221 archivoReporte << "Total pagado en intereses: $" << std::fixed << std::setprecision(2) << totalIntereses << "\n";
222 archivoReporte << "Saldo restante del préstamo: $" << std::fixed << std::setprecision(2) << saldoRestante << "\n";
223
224 archivoReporte.close();
225}

235void Prestamo::generarTablaAmortizacionTxt(const std::string& nombreArchivo) const {
236 std::ofstream archivoSalida(nombreArchivo);

246 archivoSalida << "Cuota\tPago Mensual\tIntereses\tCapital\tSaldo Restante\n";
247
248 for (int i = 1; i <= cantidadCuotas; ++i) {
249 double intereses = saldo * (tasaInteres / 12 / 100);
250 double capital = cuotaMensual - intereses;
251 saldo -= capital;
252 totalInteres += intereses;
253
254 archivoSalida << i << "\t"
255 << std::fixed << std::setprecision(2) << cuotaMensual << "\t"
256 << std::fixed << std::setprecision(2) << intereses << "\t"
257 << std::fixed << std::setprecision(2) << capital << "\t"
258 << std::fixed << std::setprecision(2) << saldo << "\n";
259 }
260
261 archivoSalida << "Total de intereses pagados: " << std::fixed << std::setprecision(2) << totalInteres << "\n";
262 archivoSalida.close();
263}
